Функция ОбработчикКоманд(Айди, Имя, ТекстСообщения) Экспорт
	Если ТекстСообщения = "/start" Тогда
		НайтиКод = Справочники.Пользователи.НайтиПоКоду(Айди);	
		Если НайтиКод = Справочники.Пользователи.ПустаяСсылка() Тогда
			НовыйЭлемент = Справочники.Пользователи.СоздатьЭлемент();
			НовыйЭлемент.Код = Айди;
			НовыйЭлемент.Наименование = Имя;
			НовыйЭлемент.ЕстьДоступ = Истина;
			НовыйЭлемент.Записать();
			Обработка.отправитьСообщениеТГ(Айди, "Вы добавлены в базу");
			Обработка.отправитьСообщениеТГ(Айди, "Отправьте свой номер телефона используя /number")
		Иначе
			Обработка.отправитьСообщениеТГ(Айди, "Вы уже в базе");
		КонецЕсли;
		
	КонецЕсли;
	Если ТекстСообщения = "/number" Тогда
		Обработка.отправитьКнопкуТГ(Айди, Имя + ", отправьте контакт");
		//Обработка.отправитьКнопкуСообщениемТГ(Айди, "Отправьте");
		
	КонецЕсли;
