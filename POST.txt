Функция TelegramRPOST(Запрос)
	Ответ = Новый HTTPСервисОтвет(200);
	Строка = Запрос.ПолучитьТелоКакСтроку();   // В запросе содержатся данные в формате JSON
	ЧтениеJSON = Новый ЧтениеJSON();
	ЧтениеJSON.УстановитьСтроку(Строка);	
	ОтветОбъект = ПрочитатьJSON(ЧтениеJSON);
	Сообщение = ОтветОбъект.message;
	Айди = Сообщение.chat.id;
	Имя = Сообщение.chat.first_name;
	Если  Сообщение.Свойство("text") = Истина Тогда
		ТекстСообщения = Сообщение.text;
		Команды.ОбработчикКоманд(Айди, Имя, ТекстСообщения);
	ИначеЕсли Сообщение.Свойство("contact") Тогда
		НомерТелефона = Сообщение.contact.phone_number;
		Обработка.отправитьСообщениеТГ(Айди, НомерТелефона);
		Команды.ДобавитьНомерВБазу(Айди, НомерТелефона);

	КонецЕсли;

	Возврат Ответ;
